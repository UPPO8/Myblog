# 1„ÄÅÂÆö‰πâÊï∞ÊçÆÈõÜ
# ÂÆö‰πâ‰∏â‰∏™ÁÇπÈõÜ

    np.linspace(axis[0], axis[1], 100)  # yËΩ¥‰∏äÁöÑÂùáÂåÄÁöÑÁÇπ

plt.scatter(np_train_data[np_train_label == 2, 0], np_train_data[np_train_label == 2, 1], marker="s")
