def fibo(n):
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0,1]
    else:
        fibo_list = fibo(n-1)
        fibo_list.append(fibo_list[-1]+fibo_list[-2])
        return fibo_list

def double_sort(arr):
    for i in range(1,len(arr)):
        for j in range(0,len(arr)-i):
            if arr[j]>arr[j+1]:
                arr[j],arr[j+1]=arr[j+1],arr[j]
    return arr

def selection_sort(arr):
    for i in range(len(arr)):
        min_id = i
        for j in range(i+1,len(arr)):
            if arr[min_id]>arr[j]:
                min_id = j
        #Â∞ÜÊâæÂà∞ÁöÑÊúÄÂ∞èÂÖÉÁ¥†‰∏éÁ¨¨‰∏Ä‰∏™ÂÖÉÁ¥†‰∫§Êç¢

    return arr

def insertion_sort(arr):
    for i in range(1,len(arr)):
        key = arr[i]
        j = i-1
        # j‰∏çËÉΩ‰∏∫Ë¥üÊï∞
        while j >= 0 and key < arr[j]:
            arr[j+1] = arr[j]
            j -= 1
        arr[j+1] = key
    return arr

def shell_sort(arr):
    # ÂèñÊï¥ËÆ°ÁÆóÂ¢ûÈáèÔºàÈó¥ÈöîÔºâÂÄº
    gap = len(arr) // 2
    while gap > 0:
        # ‰ªéÂ¢ûÈáèÂÄºÂºÄÂßãÈÅçÂéÜÊØîËæÉ
        for i in range(gap, len(arr)):
            j = i
            current = arr[i]
            # ÂÖÉÁ¥†‰∏é‰ªñÂêåÂàóÁöÑÂâçÈù¢ÁöÑÊØè‰∏™ÂÖÉÁ¥†ÊØîËæÉÔºåÂ¶ÇÊûúÊØîÂâçÈù¢ÁöÑÂ∞èÂàô‰∫íÊç¢
            while j - gap >= 0 and current < arr[j - gap]:
                arr[j] = arr[j - gap]
                j -= gap
            arr[j] = current
        # Áº©Â∞èÂ¢ûÈáèÔºàÈó¥ÈöîÔºâÂÄº
        gap //= 2
    return arr

def quick_sort(arr):
    if len(arr) < 2:
        return arr
    # ÈÄâÂèñÂü∫ÂáÜÔºåÈöè‰æøÈÄâÂì™‰∏™ÈÉΩÂèØ‰ª•ÔºåÈÄâ‰∏≠Èó¥ÁöÑ‰æø‰∫éÁêÜËß£
    mid = arr[len(arr) // 2]
    # ÂÆö‰πâÂü∫ÂáÜÂÄºÂ∑¶Âè≥‰∏§‰∏™Êï∞Âàó
    left, right = [], []
    # ‰ªéÂéüÂßãÊï∞ÁªÑ‰∏≠ÁßªÈô§Âü∫ÂáÜÂÄº
    arr.remove(mid)
    for item in arr:
        # Â§ß‰∫éÂü∫ÂáÜÂÄºÊîæÂè≥Ëæπ
        if item >= mid:
            right.append(item)
        else:
            # Â∞è‰∫éÂü∫ÂáÜÂÄºÊîæÂ∑¶Ëæπ
            left.append(item)
    # ‰ΩøÁî®Ëø≠‰ª£ËøõË°åÊØîËæÉ
    return quick_sort(left) + [mid] + quick_sort(right)

def merge_sort(arr):
    """ÂΩíÂπ∂ÊéíÂ∫è"""
    if len(arr) == 1:
        return arr
    # ‰ΩøÁî®‰∫åÂàÜÊ≥ïÂ∞ÜÊï∞ÂàóÂàÜ‰∏§‰∏™
    mid = len(arr) // 2
    left = arr[:mid]
    right = arr[mid:]
    # ‰ΩøÁî®ÈÄíÂΩíËøêÁÆó
    return marge(merge_sort(left), merge_sort(right))


def marge(left, right):
    """ÊéíÂ∫èÂêàÂπ∂‰∏§‰∏™Êï∞Âàó"""
    result = []
    # ‰∏§‰∏™Êï∞ÂàóÈÉΩÊúâÂÄº
    while len(left) > 0 and len(right) > 0:
        # Â∑¶Âè≥‰∏§‰∏™Êï∞ÂàóÁ¨¨‰∏Ä‰∏™ÊúÄÂ∞èÊîæÂâçÈù¢
        if left[0] <= right[0]:
            result.append(left.pop(0))
        else:
            result.append(right.pop(0))
    # Âè™Êúâ‰∏Ä‰∏™Êï∞Âàó‰∏≠ËøòÊúâÂÄºÔºåÁõ¥Êé•Ê∑ªÂä†
    result += left
    result += right
    return result

from heapq import merge

def merge_sort(lst):
    if len(lst) <= 1:
        return lst          # ‰ªéÈÄíÂΩí‰∏≠ËøîÂõûÈïøÂ∫¶‰∏∫1ÁöÑÂ∫èÂàó
    middle = len(lst) // 2

    right = merge_sort(lst[middle:])
    return list(merge(left, right))
