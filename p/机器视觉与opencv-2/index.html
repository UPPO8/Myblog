import cv2
from matplotlib import pyplot as plt

if __name__ == "__main__":
    path = "./006.png"
    image_np = cv2.imread(path)
    # ËΩ¨‰∏∫ÁÅ∞Â∫¶Âõæ
    image_np_gray = cv2.cvtColor(image_np, cv2.COLOR_BGR2GRAY)
    # ËøõË°åËá™ÈÄÇÂ∫î‰∫åÂÄºÂåñ
    image_np_thresh = cv2.adaptiveThreshold(
        image_np_gray, 255,
        cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
        cv2.THRESH_BINARY, 11, 3)

    kernel3 = cv2.getStructuringElement(cv2.MORPH_CROSS, (3, 3))
    kernel4 = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
    kernel5 = cv2.getStructuringElement(cv2.MORPH_RECT, (7, 7))

    erode_MORPH_CROSS = cv2.erode(image_np_thresh, kernel3)
    #‰∏çÂêåÂç∑ÁßØÊ†∏Â§ßÂ∞èÂØπÊØî
    kernel_size_3 = cv2.erode(image_np_thresh, kernel1)
    kernel_size_5 = cv2.erode(image_np_thresh, kernel4)

    # ËøîÂõûÂ§ÑÁêÜÊ≠£Á°ÆÂêéÁöÑÂÜÖÂÆπ
    titles = ['MORPH_RECT','MORPH_ELLIPSE','MORPH_CROSS',
              'kernel_size_3','kernel_size_5','kernel_size_7']
    images = [erode_MORPH_RECT,erode_MORPH_ELLIPSE,erode_MORPH_CROSS,
              kernel_size_3,kernel_size_5,kernel_size_7]
    for i in range(6):
        plt.subplot(2,3,i+1),
        plt.imshow(images[i], 'gray')
        plt.title(titles[i])
        plt.xticks([])
        plt.yticks([])
    plt.show()

def rotate(image, angle, center = None, scale = 1.0):

    (h, w) = image.shape[:2]

    if center is None:
        center = (w / 2, h / 2)

    M = cv2.getRotationMatrix2D(center, angle, scale)
    rotated = cv2.warpAffine(image, M, (w, h))

    return rotated
